亲测有效。中文输入状态下是1025（不是[语言区域代码LCID](2021-12-30-Windows%20语言代码标识符(LCID).md)），英文输入状态是0，和CAPSLOCK键的状态没有关系。

## 生产环境代码
点击鼠标左键可以显示目前中英文输入状态。

```autohotkey
#NoEnv
#Persistent,On
#SingleInstance,Force
SendMode, Input

~LButton::
    ToolTip
    If (IME_GetConvMode()=0)
        ToolTip,EN 
    else
        ToolTip,中
return

~Lbutton up::
    Sleep,500
    ToolTip
return


GetGUIThreadInfo_hwndActive(WinTitle="A") {
    ControlGet, hwnd, HWND,,, %WinTitle%
    if (WinActive(WinTitle)) {
    ptrSize := !A_PtrSize ? 4 : A_PtrSize
    VarSetCapacity(stGTI, cbSize:=4+4+(PtrSize*6)+16, 0)
    NumPut(cbSize, stGTI,  0, "UInt")
    return hwnd := DllCall("GetGUIThreadInfo", "Uint", 0, "Ptr", &stGTI)
    ? NumGet(stGTI, 8+PtrSize, "Ptr") : hwnd
    }
    else {
    return hwnd
    }
}

IME_GetConvMode(WinTitle="A")   {
hwnd :=GetGUIThreadInfo_hwndActive(WinTitle)
return DllCall("SendMessage"
, "Ptr", DllCall("imm32\ImmGetDefaultIMEWnd", "Ptr", hwnd)
, "UInt", 0x0283
,  "Int", 0x001
,  "Int", 0) & 0xffff
}


```
